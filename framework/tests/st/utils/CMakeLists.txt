            include
            ${PTO_FWK_SRC_ROOT}/framework/src/machine/runtime
            ${PTO_FWK_SRC_ROOT}/framework/src/cost_model
        INTERFACE
            $<BUILD_INTERFACE:$<$<BOOL:${BUILD_WITH_CANN}>:${ASCEND_CANN_PACKAGE_PATH}/include>>
